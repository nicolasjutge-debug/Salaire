function primeExperienceMensuelle(annee, mois, config) { const base = Number(config.baseConventionnelleClasse3) || 2089.17; const key = annee + "-" + String(mois).padStart(2, "0"); const dateValorisation = config.valorisationConventionDate || "2025-08-01"; const dateHausse2026 = "2026-08-01"; let taux; if (key < dateValorisation.slice(0, 7)) { taux = Number(config.valorisationTauxDepart) || 9; } else if (key < dateHausse2026.slice(0, 7)) { taux = Number(config.valorisationTauxDepart) || 9; } else { taux = 10.5; } return { base: base, taux: taux, montant: round2(base * taux / 100) }; }
